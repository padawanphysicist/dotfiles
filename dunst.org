[[https://dunst-project.org/][Dunst]] is a lightweight and customizable notification daemon. My configuration is structured in three parts:

* Keyboard shortcuts

#+begin_src conf
[shortcuts]
close = ctrl+space
close_all = ctrl+shift+space
context = ctrl+shift+period
#+end_src

* Global settings

#+begin_src conf
[global]
font = "SFNS Display 14"
# Allow html markup
markup = yes
# Format of the notifications
# %s - summary
# %p - progress
# %b - body of the message
format = "<b>%s %p</b>\n%b"
# Sort messages by priority
sort = yes
# Show how many notifications are hidden
indicate_hidden = true
idle_threshold = 0
# Size of the notification
geometry = "300x5-20+20"
alignment = center
show_age_threshold = 60
sticky_history = yes
follow = mouse
word_wrap = yes
separator_height = 2
padding = 10
horizontal_padding = 10
separator_color = frame
startup_notification = true
frame_width = 3
frame_color = "#6092BE"
#+end_src

* Colors

#+begin_src conf
[urgency_low]
background = "#ffffff"
foreground = "#000000"
timeout = 5

[urgency_normal]
background = "#94DBFF"
foreground = "#000000"
timeout = 10

[urgency_critical]
background = "#ff9999"
foreground = "#000000"
timeout = 0
